📊 SQL 핵심 명령어 및 문법 요약

        
아래 내용은 시험에 자주 나오는 SQL 문법을 기능 설명 + 예시문 형태로 정리한 것입니다.


        

          
            
명령어 / 키워드기능 설명문법 및 활용 예시

          
          
            

              SELECT
              조회할 열이나 계산식을 지정하는 가장 기본 명령어
              
SELECT bookname, price FROM Book;
SELECT * FROM Customer;

            

            

              FROM
              데이터를 가져올 테이블을 지정
              
SELECT *
FROM Book;

            

            

              *
              테이블의 모든 열을 조회할 때 사용
              
SELECT * FROM Orders;

            

            

              컬럼명
              필요한 열만 골라 조회. 예: bookname, price, name, orderdate
              
SELECT bookname FROM Book;
SELECT name FROM Customer;
SELECT saleprice FROM Orders;

            

            

              문자열/날짜 값
              문자와 날짜는 작은따옴표로 감싸서 조건에 사용
              
WHERE publisher = '굿스포츠'
WHERE orderdate = '2020-07-03'

            

            

              비교 연산자
              =, !=, <>, >, >=, <, <=로 값을 비교
              
WHERE price >= 10000;
WHERE saleprice < 10000;
WHERE phone != '5678';

            

            

              AND
              여러 조건을 모두 만족하는 행만 선택
              
WHERE publisher = '굿스포츠'
  AND price >= 7000;

            

            

              BETWEEN A AND B
              A 이상 B 이하의 범위에 포함되는 값을 검색
              
WHERE price BETWEEN 10000 AND 20000;
WHERE saleprice BETWEEN 7000 AND 13000;

            

            

              IS NULL
              값이 비어 있는 행을 찾음. NULL은 = NULL로 비교하지 않음
              
WHERE phone IS NULL;

            

            

              IS NOT NULL
              NULL이 아닌 값이 들어 있는 행을 찾음
              
WHERE phone IS NOT NULL;

            

            

              COUNT(*)
              전체 행의 개수를 셈
              
SELECT COUNT(*) FROM Book;

            

            

              SUM()
              숫자 열의 합계를 구함
              
SELECT SUM(saleprice) FROM Orders;

            

            

              AVG()
              숫자 열의 평균을 구함
              
SELECT AVG(price) FROM Book;

            

            

              MIN()
              가장 작은 값 또는 가장 이른 날짜를 구함
              
SELECT MIN(price) FROM Book;
SELECT MIN(orderdate) FROM Orders;

            

            

              MAX()
              가장 큰 값 또는 가장 최근 날짜를 구함
              
SELECT MAX(price) FROM Book;
SELECT MAX(orderdate) FROM Orders;

            

            

              LIMIT
              조회 결과에서 앞의 N개만 출력. 정렬과 함께 상위/하위 1개 찾기에 자주 사용
              
SELECT * FROM Book
ORDER BY price DESC
LIMIT 1;

            

            

              JOIN / INNER JOIN
              두 테이블에서 연결 조건이 서로 맞는 행만 합쳐 조회. MySQL에서 JOIN은 보통 INNER JOIN과 같은 의미
              
FROM Customer JOIN Orders
ON Customer.custid = Orders.custid;

            

            

              계산식
              열끼리 더하기, 빼기 등 계산한 결과를 조회하거나 조건에 사용
              
SELECT Book.price - Orders.saleprice AS discount
FROM Book JOIN Orders
ON Book.bookid = Orders.bookid;

            

            

              서브쿼리 (SELECT ...)
              쿼리 안에 들어가는 또 다른 조회문. 평균, 최고값 등과 비교할 때 사용
              
WHERE price > (SELECT AVG(price) FROM Book);

            

            

              DISTINCT
              조회 결과에서 중복된 행을 제거하고 고유한 값만 출력
              
SELECT DISTINCT publisher FROM Book;

            

            

              COUNT(DISTINCT ...)
              중복을 제거한 데이터의 순수한 종류 개수를 구함
              
SELECT COUNT(DISTINCT custid) FROM Orders;

            

            

              WHERE
              테이블의 개별 행(Row)에 조건을 걸어 그룹화하기 전에 필터링
              
WHERE custid >= 3;
WHERE phone != '5678';

            

            

              GROUP BY
              특정 열의 값이 같은 데이터들을 하나의 그룹으로 묶음
              
SELECT custid, COUNT(*)
FROM Orders
GROUP BY custid;

            

            

              HAVING
              GROUP BY로 그룹화된 결과, 즉 통계값에 조건을 걸어 필터링
              
SELECT custid, COUNT(*)
FROM Orders
GROUP BY custid
HAVING COUNT(*) >= 3;

            

            

              LIKE
              와일드카드 기호 _, %를 사용하여 문자열 패턴 검색
              
WHERE name LIKE '_대%';
-- 두 번째 글자가 '대'

WHERE bookname LIKE '%축구%';
-- '축구' 포함

            

            

              NOT LIKE
              특정 문자열 패턴과 일치하지 않는 행을 선택
              
WHERE bookname NOT LIKE '%축구%';

            

            

              OR
              나열된 조건 중 하나라도 만족하면 데이터를 선택
              
WHERE publisher = '굿스포츠'
   OR publisher = '대한미디어';

            

            

              IN
              여러 값 목록 중 하나와 일치하는 데이터를 선택. OR 조건을 짧게 쓸 때 자주 사용
              
SELECT * FROM Book
WHERE publisher IN ('굿스포츠', '대한미디어');

-- 아래 OR 조건과 비슷한 의미
WHERE publisher = '굿스포츠'
   OR publisher = '대한미디어';

            

            

              NOT IN
              목록이나 서브쿼리 결과에 포함되지 않는 행을 선택
              
SELECT * FROM Book
WHERE bookid NOT IN (SELECT bookid FROM Orders);

            

            

              ON
              JOIN할 때 두 테이블을 어떤 컬럼 기준으로 연결할지 지정하는 조건
              
FROM Customer JOIN Orders
ON Customer.custid = Orders.custid;

FROM Book JOIN Orders
ON Book.bookid = Orders.bookid;

            

            

              ORDER BY
              데이터를 특정 열 기준으로 정렬. 보통 쿼리문의 가장 마지막에 위치
              
ORDER BY price ASC;
ORDER BY orderdate DESC;

            

            

              DESC
              내림차순 정렬. 큰 값에서 작은 값 순서
              
ORDER BY orderid DESC;
-- 10 → 1

            

            

              ASC
              오름차순 정렬. 작은 값에서 큰 값 순서. 기본값이라 생략 가능
              
ORDER BY orderid ASC;
-- 1 → 10

            

            

              복수 정렬 ,
              1순위 정렬 값이 같을 때 쉼표 뒤의 2순위 기준으로 재정렬
              
ORDER BY orderdate DESC, saleprice ASC;

            

            

              LEFT OUTER JOIN
              왼쪽 테이블을 기준으로 합침. 오른쪽에 짝이 없어도 왼쪽 데이터는 다 살림
              
FROM Book LEFT OUTER JOIN Orders
ON Book.bookid = Orders.bookid;

            

            

              RIGHT OUTER JOIN
              오른쪽 테이블을 기준으로 합침. 왼쪽에 짝이 없어도 오른쪽 데이터는 다 살림
              
FROM Book RIGHT OUTER JOIN Orders
ON Book.bookid = Orders.bookid;

            

            

              UNION
              두 개의 조회 결과를 위아래로 합치는 합집합. 중복 데이터는 자동 제거
              
SELECT name FROM Customer
UNION
SELECT name FROM NewCustomer;

            

            

              INTERSECT
              두 개의 조회 결과에서 공통으로 존재하는 데이터만 뽑는 교집합
              
SELECT name FROM CustomerA
INTERSECT
SELECT name FROM CustomerB;

            

            

              AS
              열 이름이나 테이블 이름에 별명(Alias)을 붙여 일시적으로 바꿔줌
              
SELECT name AS 고객이름 FROM Customer;
SELECT SUM(saleprice) AS 총액 FROM Orders;

            

            

              EXISTS
              서브쿼리의 조건을 만족하는 데이터가 하나라도 존재하는지 참/거짓 확인
              
SELECT * FROM Customer
WHERE EXISTS (
  SELECT * FROM Orders
  WHERE Orders.custid = Customer.custid
);

            

            

              NOT EXISTS
              서브쿼리 조건을 만족하는 데이터가 하나도 없는 행을 선택
              
SELECT * FROM Customer
WHERE NOT EXISTS (
  SELECT * FROM Orders
  WHERE Orders.custid = Customer.custid
);

            

          
        

      


      

        
⚠️ 핵심 주의사항 — 시험 단골 출제


        
1. WHERE vs HAVING 실행 순서

        
컴퓨터는 WHERE로 개별 데이터를 먼저 거른 뒤, GROUP BY로 묶고, 마지막에 HAVING으로 그룹 결과를 검사합니다.

        
SELECT custid, COUNT(*)
FROM Orders
WHERE saleprice >= 10000
GROUP BY custid
HAVING COUNT(*) >= 2;

        
    
          
  • WHERE: 개별 행 조건
    • 
          
  • HAVING: COUNT, SUM, AVG 같은 집계 결과 조건
    • 
          
  • 주의: WHERE COUNT(*) >= 2처럼 쓰면 안 됩니다.
    • 
        


        
2. NULL 비교 연산

        
NULL은 값이 없다는 뜻이라 =, !=, <>로 정확히 비교할 수 없습니다.

        
-- 올바른 NULL 비교
WHERE phone IS NULL
WHERE phone IS NOT NULL

        
!=이나 <>로 데이터를 거를 때, 값이 NULL인 행은 비교 대상에서 제외될 수 있습니다. NULL까지 포함하려면 다음처럼 씁니다.

        
WHERE phone != '5678' OR phone IS NULL;


        
3. LIKE 와일드카드 구분

        
    
          
  • _: 정확히 1글자 자리
    • 
          
  • %: 글자 수 제한 없음
    • 
        

        
WHERE name LIKE '_대%';
-- 두 번째 글자가 '대'

WHERE name LIKE '__대%';
-- 세 번째 글자가 '대'

WHERE bookname LIKE '%축구%';
-- 앞뒤 글자 수 상관없이 '축구' 포함

# SQL 실행 순서 기억하기
FROM / JOIN
→ WHERE
→ GROUP BY
→ HAVING
→ SELECT
→ ORDER BY
→ LIMIT

# WHERE vs HAVING
- WHERE: 그룹 전, 개별 행 조건
- HAVING: 그룹 후, 집계 결과 조건

# LIKE
- '박%'   : 박으로 시작
- '%축구%' : 축구 포함
- '_대%'  : 두 번째 글자가 대

# IN / ON
- IN : 여러 값 중 하나인지 확인
  WHERE publisher IN ('굿스포츠', '대한미디어')
- ON : JOIN 연결 조건
  ON Customer.custid = Orders.custid

# OUTER JOIN
- LEFT OUTER JOIN  : 왼쪽 테이블은 전부 살림
- RIGHT OUTER JOIN : 오른쪽 테이블은 전부 살림

# 집계 함수
- COUNT(*) : 행 개수
- SUM(열)  : 합계
- AVG(열)  : 평균
- MIN(열)  : 최솟값
- MAX(열)  : 최댓값

# 범위 / 제한 / 제외
- BETWEEN A AND B : A 이상 B 이하
- LIMIT 1         : 결과 1개만 보기
- NOT LIKE        : 패턴 제외
- NOT IN          : 목록/서브쿼리에 없는 값
- NOT EXISTS      : 관련 데이터가 없는 경우

# JOIN과 서브쿼리
- JOIN 또는 INNER JOIN : 양쪽에 연결되는 데이터만 합침
- 서브쿼리: WHERE price > (SELECT AVG(price) FROM Book)
- 계산식: Book.price - Orders.saleprice AS discount
