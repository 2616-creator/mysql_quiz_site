MYSQL

        
📊 SQL 핵심 명령어 및 문법 요약

        
아래 내용은 시험에 자주 나오는 SQL 문법을 기능 설명 + 예시문 형태로 정리한 것입니다.


        

          
            
명령어 / 키워드기능 설명문법 및 활용 예시

          
          
            

              DISTINCT
              조회 결과에서 중복된 행을 제거하고 고유한 값만 출력
              
SELECT DISTINCT publisher FROM Book;

            

            

              COUNT(DISTINCT ...)
              중복을 제거한 데이터의 순수한 종류 개수를 구함
              
SELECT COUNT(DISTINCT custid) FROM Orders;

            

            

              WHERE
              테이블의 개별 행(Row)에 조건을 걸어 그룹화하기 전에 필터링
              
WHERE custid >= 3;
WHERE phone != '5678';

            

            

              GROUP BY
              특정 열의 값이 같은 데이터들을 하나의 그룹으로 묶음
              
SELECT custid, COUNT(*)
FROM Orders
GROUP BY custid;

            

            

              HAVING
              GROUP BY로 그룹화된 결과, 즉 통계값에 조건을 걸어 필터링
              
SELECT custid, COUNT(*)
FROM Orders
GROUP BY custid
HAVING COUNT(*) >= 3;

            

            

              LIKE
              와일드카드 기호 _, %를 사용하여 문자열 패턴 검색
              
WHERE name LIKE '_대%';
-- 두 번째 글자가 '대'

WHERE bookname LIKE '%축구%';
-- '축구' 포함

            

            

              OR
              나열된 조건 중 하나라도 만족하면 데이터를 선택
              
WHERE publisher = '굿스포츠'
   OR publisher = '대한미디어';

            

            

              ORDER BY
              데이터를 특정 열 기준으로 정렬. 보통 쿼리문의 가장 마지막에 위치
              
ORDER BY price ASC;
ORDER BY orderdate DESC;

            

            

              DESC
              내림차순 정렬. 큰 값에서 작은 값 순서
              
ORDER BY orderid DESC;
-- 10 → 1

            

            

              ASC
              오름차순 정렬. 작은 값에서 큰 값 순서. 기본값이라 생략 가능
              
ORDER BY orderid ASC;
-- 1 → 10

            

            

              복수 정렬 ,
              1순위 정렬 값이 같을 때 쉼표 뒤의 2순위 기준으로 재정렬
              
ORDER BY orderdate DESC, saleprice ASC;

            

            

              LEFT OUTER JOIN
              왼쪽 테이블을 기준으로 합침. 오른쪽에 짝이 없어도 왼쪽 데이터는 다 살림
              
FROM Book LEFT OUTER JOIN Orders
ON Book.bookid = Orders.bookid;

            

            

              RIGHT OUTER JOIN
              오른쪽 테이블을 기준으로 합침. 왼쪽에 짝이 없어도 오른쪽 데이터는 다 살림
              
FROM Book RIGHT OUTER JOIN Orders
ON Book.bookid = Orders.bookid;

            

            

              UNION
              두 개의 조회 결과를 위아래로 합치는 합집합. 중복 데이터는 자동 제거
              
SELECT name FROM Customer
UNION
SELECT name FROM NewCustomer;

            

            

              INTERSECT
              두 개의 조회 결과에서 공통으로 존재하는 데이터만 뽑는 교집합
              
SELECT name FROM CustomerA
INTERSECT
SELECT name FROM CustomerB;

            

            

              AS
              열 이름이나 테이블 이름에 별명(Alias)을 붙여 일시적으로 바꿔줌
              
SELECT name AS 고객이름 FROM Customer;
SELECT SUM(saleprice) AS 총액 FROM Orders;

            

            

              EXISTS
              서브쿼리의 조건을 만족하는 데이터가 하나라도 존재하는지 참/거짓 확인
              
SELECT * FROM Customer
WHERE EXISTS (
  SELECT * FROM Orders
  WHERE Orders.custid = Customer.custid
);

            

          
        

      


      

        
⚠️ 핵심 주의사항 — 시험 단골 출제


        
1. WHERE vs HAVING 실행 순서

        
컴퓨터는 WHERE로 개별 데이터를 먼저 거른 뒤, GROUP BY로 묶고, 마지막에 HAVING으로 그룹 결과를 검사합니다.

        
SELECT custid, COUNT(*)
FROM Orders
WHERE saleprice >= 10000
GROUP BY custid
HAVING COUNT(*) >= 2;

        
    
          
  • WHERE: 개별 행 조건
    • 
          
  • HAVING: COUNT, SUM, AVG 같은 집계 결과 조건
    • 
          
  • 주의: WHERE COUNT(*) >= 2처럼 쓰면 안 됩니다.
    • 
        


        
2. NULL 비교 연산

        
NULL은 값이 없다는 뜻이라 =, !=, <>로 정확히 비교할 수 없습니다.

        
-- 올바른 NULL 비교
WHERE phone IS NULL
WHERE phone IS NOT NULL

        
!=이나 <>로 데이터를 거를 때, 값이 NULL인 행은 비교 대상에서 제외될 수 있습니다. NULL까지 포함하려면 다음처럼 씁니다.

        
WHERE phone != '5678' OR phone IS NULL;


        
3. LIKE 와일드카드 구분

        
    
          
  • _: 정확히 1글자 자리
    • 
          
  • %: 글자 수 제한 없음
    • 
        

        
WHERE name LIKE '_대%';
-- 두 번째 글자가 '대'

WHERE name LIKE '__대%';
-- 세 번째 글자가 '대'

WHERE bookname LIKE '%축구%';
-- 앞뒤 글자 수 상관없이 '축구' 포함

# SQL 실행 순서 기억하기
FROM / JOIN
→ WHERE
→ GROUP BY
→ HAVING
→ SELECT
→ ORDER BY
→ LIMIT

# WHERE vs HAVING
- WHERE: 그룹 전, 개별 행 조건
- HAVING: 그룹 후, 집계 결과 조건

# LIKE
- '박%'   : 박으로 시작
- '%축구%' : 축구 포함
- '_대%'  : 두 번째 글자가 대

# OUTER JOIN
- LEFT OUTER JOIN  : 왼쪽 테이블은 전부 살림
- RIGHT OUTER JOIN : 오른쪽 테이블은 전부 살림
